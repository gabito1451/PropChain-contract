60    #[ink(storage)]
61    pub struct AnalyticsDashboard {
62        /// Administrator of the analytics dashboard
63        admin: AccountId,
64        /// Current market metrics
65        current_metrics: MarketMetrics,
66        /// Historical market trends
67        historical_trends: ink::storage::Mapping<u64, MarketTrend>,
68        /// Trend count
69        trend_count: u64,
70    }
71
72    impl AnalyticsDashboard {
73        #[ink(constructor)]
74        pub fn new() -> Self {
75            let caller = Self::env().caller();
76            Self {
77                admin: caller,
78                current_metrics: MarketMetrics {
79                    average_price: 0,
80                    total_volume: 0,
81                    properties_listed: 0,
82                },
83                historical_trends: ink::storage::Mapping::default(),
84                trend_count: 0,
85            }
86        }
87
88        /// Implement property market metrics calculation (average price, volume, etc.)
89        #[ink(message)]
90        pub fn get_market_metrics(&self) -> MarketMetrics {
91            self.current_metrics.clone()
92        }
93
94        #[ink(message)]
95        pub fn update_market_metrics(&mut self, average_price: u128, total_volume: u128, properties_listed: u64) {
96            self.ensure_admin();
97            self.current_metrics = MarketMetrics {
98                average_price,
99                total_volume,
100                properties_listed,
101            };
102        }
103
104        /// Create market trend analysis with historical data
105        #[ink(message)]
106        pub fn add_market_trend(&mut self, trend: MarketTrend) {
107            self.ensure_admin();
108            self.historical_trends.insert(self.trend_count, &trend);
109            self.trend_count += 1;
110        }
111
112        #[ink(message)]
113        pub fn get_historical_trends(&self) -> Vec<MarketTrend> {
114            let mut trends = Vec::new();
115            for i in 0..self.trend_count {
116                if let Some(trend) = self.historical_trends.get(i) {
117                    trends.push(trend);
118                }
119            }
120            trends
121        }
122
123        /// Create automated market reports generation
124        #[ink(message)]
125        pub fn generate_market_report(&self) -> MarketReport {
126            let latest_trend = if self.trend_count > 0 {
127                self.historical_trends.get(self.trend_count - 1).unwrap_or(MarketTrend {
128                    period_start: 0,
129                    period_end: 0,
130                    price_change_percentage: 0,
131                    volume_change_percentage: 0,
132                })
133            } else {
134                MarketTrend {
135                    period_start: 0,
136                    period_end: 0,
137                    price_change_percentage: 0,
138                    volume_change_percentage: 0,
139                }
140            };
141
142            MarketReport {
143                generated_at: self.env().block_timestamp(),
144                metrics: self.current_metrics.clone(),
145                trend: latest_trend,
146                insights: String::from("Market is relatively stable. Gas optimization is recommended."),
147            }
148        }
149
150        /// Add gas usage optimization recommendations
151        #[ink(message)]
152        pub fn get_gas_optimization_recommendations(&self) -> String {
153            String::from("Use batched operations and limit nested looping over dynamic collections (e.g. vectors). Store large items in Mappings instead of Vecs.")
154        }
155
156        /// Ensure only the admin can modify metrics
157        fn ensure_admin(&self) {
158            assert_eq!(self.env().caller(), self.admin, "Unauthorized: Analytics admin only");
159        }
160    }

166        #[ink::test]
167        fn market_metrics_defaults() {
168            let contract = AnalyticsDashboard::new();
169            let metrics = contract.get_market_metrics();

175        #[ink::test]
176        fn update_market_metrics_works() {
177            let mut contract = AnalyticsDashboard::new();
178            contract.update_market_metrics(1000, 5000, 10);
179            let metrics = contract.get_market_metrics();
180            assert_eq!(metrics.average_price, 1000);
181            assert_eq!(metrics.total_volume, 5000);
182            assert_eq!(metrics.properties_listed, 10);
183        }
184
185        #[ink::test]
186        fn add_market_trend_works() {
187            let mut contract = AnalyticsDashboard::new();
188            let trend = MarketTrend {
189                period_start: 100,
190                period_end: 200,
191                price_change_percentage: 5,
192                volume_change_percentage: 10,
193            };
194            contract.add_market_trend(trend.clone());
195            let trends = contract.get_historical_trends();
196            assert_eq!(trends.len(), 1);
197            assert_eq!(trends[0].price_change_percentage, 5);
198        }
199
200        #[ink::test]
201        fn generate_market_report_works() {
202            let contract = AnalyticsDashboard::new();
203            let report = contract.generate_market_report();
204            assert_eq!(report.metrics.average_price, 0);
205            assert!(report.insights.contains("Gas optimization"));
206        }
